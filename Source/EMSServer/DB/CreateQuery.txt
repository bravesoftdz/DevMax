CREATE TABLE DMX_VIEW(
	VIEW_SEQ 	INTEGER NOT NULL, 
	DESCRIPTION	BLOB SUB_TYPE TEXT SEGMENT SIZE 80,
	MAIN_PAGE_SEQ	INTEGER NOT NULL, 
CONSTRAINT "VIEW_ID_PK" PRIMARY KEY ("VIEW_SEQ")
);

CREATE GENERATOR "VIEW_SEQ_GEN";

CREATE TRIGGER SET_VIEW_SEQ FOR DMX_VIEW
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN
    new.VIEW_SEQ = gen_id(VIEW_SEQ_GEN, 1);
END;
